#include <cctype>
#include <iostream>
using namespace std;

void func(string s) {
  for (int i = 0; i < 26; ++i) {
    string temp(s);
    for (auto &c : temp) {
      c = tolower(c);
      c = ((c + i) - 97) % 26 + 97;
    }
    cout << temp << endl;
  }
}

int main() {
  string s;
  cin >> s;
  func(s);
}
